### 3. Текст программы
```c++
#include <iostream>
using namespace std;
    int main()
{
    int a, b, c, x, y, d, f;
    bool r;
    do {
        cout << "Made by Salikov Viktor" << endl;
        cout << "Laba #1" << endl;
        cout << "Enter the first x\n";
        cin >> d;
        cout << "Enter the Last x\n";
        cin >> b;
        cout << "Enter number of lines\n";
        cin >> c;
        f = (b - d) /( c-1) ;
        cout << "x y" << endl;
        for (int i = 0; i < c; ++i)
        {
            x = d + f * i;
            y = x * x;
            cout << "x[" << i << "]=" << x << "    " << "y[" << i << "]=" << y << endl;
        }
        cout << "Enter 1 to repeat" << endl;
        cin >> a;
        r = (a == 1);
    } while (r);
    cout << "Goodbye\n";
    system("pause");
    return 0;
} 
```
